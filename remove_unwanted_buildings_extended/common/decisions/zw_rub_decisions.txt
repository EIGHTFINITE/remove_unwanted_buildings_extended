zw_remove_conscripts_decision = {

	is_shown = {
		is_player = yes
		any_scope_state = {
			has_decree = decree_zw_remove_buildings
			has_building = building_conscription_center
		}
	}

	possible = {}

	when_taken = {
		every_scope_state = {
			limit = {
				has_decree = decree_zw_remove_buildings
				has_building = building_conscription_center
			}
			remove_building = building_conscription_center
		}
	}
}
